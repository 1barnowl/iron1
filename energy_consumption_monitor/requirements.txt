"""
Database Manager - Handles all database operations using asyncpg
"""

import asyncpg
import os
import logging
from typing import List, Optional, Dict, Any
from datetime import datetime

from models import (
    SensorDataResponse,
    DeviceResponse,
    AlertResponse,
    PowerStatsResponse
)

logger = logging.getLogger(__name__)


class DatabaseManager:
    def __init__(self):
        self.database_url = os.getenv(
            "DATABASE_URL",
            "postgresql://energy_user:energy_pass@timescaledb:5432/energy_monitor"
        )
        self.pool: Optional[asyncpg.Pool] = None
    
    async def connect(self):
        """Establish database connection pool"""
        try:
            self.pool = await asyncpg.create_pool(
                self.database_url,
                min_size=5,
                max_size=20,
                command_timeout=60
            )
            logger.info("Database connection pool created")
        except Exception as e:
            logger.error(f"Failed to connect to database: {e}")
            raise
    
    async def disconnect(self):
        """Close database connection pool"""
        if self.pool:
            await self.pool.close()
            logger.info("Database connection pool closed")
    
    async def check_health(self) -> bool:
        """Check database health"""
        try:
            async with self.pool.acquire() as conn:
                await conn.fetchval("SELECT 1")
            return True
        except Exception as e:
            logger.error(f"Database health check failed: {e}")
            return False
    
    async def insert_sensor_data(self, data: Dict[str, Any]):
        """Insert sensor data into database"""
        try:
            async with self.pool.acquire() as conn:
                await conn.execute(
                    """
                    INSERT INTO sensor_data 
                    (time, device_id, location, power_watts, voltage, current, 
                     power_factor, frequency, metadata)
                    VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
                    """,
                    data.get("time", datetime.utcnow()),
                    data["device_id"],
                    data.get("location"),
                    data.get("power_watts"),
                    data.get("voltage"),
                    data.get("current"),
                    data.get("power_factor"),
                    data.get("frequency"),
                    data.get("metadata")
                )
        except Exception as e:
            logger.error(f"Error inserting sensor data: {e}")
            raise
    
    async def get_all_devices(self) -> List[DeviceResponse]:
        """Get all registered devices"""
        try:
            async with self.pool.acquire() as conn:
                rows = await conn.fetch(
                    """
                    SELECT device_id, device_name, location, device_type, 
                           rated_power, installation_date, metadata
                    FROM devices
                    ORDER BY device_name
                    """
                )
                return [DeviceResponse(**dict(row)) for row in rows]
        except Exception as e:
            logger.error(f"Error fetching devices: {e}")
            raise
    
    async def get_device(self, device_id: str) -> Optional[DeviceResponse]:
        """Get specific device"""
        try:
            async with self.pool.acquire() as conn:
                row = await conn.fetchrow(
                    """
                    SELECT device_id, device_name, location, device_type, 
                           rated_power, installation_date, metadata
                    FROM devices
                    WHERE device_id = $1
                    """,
                    device_id
                )
                return DeviceResponse(**dict(row)) if row else None
        except Exception as e:
            logger.error(f"Error fetching device: {e}")
            raise
    
    async def get_current_power(self) -> List[SensorDataResponse]:
        """Get latest power readings for all devices"""
        try:
            async with self.pool.acquire() as conn:
                rows = await conn.fetch(
                    """
                    SELECT DISTINCT ON (device_id)
                        time, device_id, location, power_watts, 
                        voltage, current, power_factor, frequency
                    FROM sensor_data
                    ORDER BY device_id, time DESC
                    """
                )
                return [SensorDataResponse(**dict(row)) for row in rows]
        except Exception as e:
            logger.error(f"Error fetching current power: {e}")
            raise
    
    async def get_sensor_history(
        self,
        device_id: str,
        start_time: datetime,
        limit: int = 1000
    ) -> List[SensorDataResponse]:
        """Get historical sensor data for a device"""
        try:
            async with self.pool.acquire() as conn:
                rows = await conn.fetch(
                    """
                    SELECT time, device_id, location, power_watts, 
                           voltage, current, power_factor, frequency
                    FROM sensor_data
                    WHERE device_id = $1 AND time >= $2
                    ORDER BY time DESC
                    LIMIT $3
                    """,
                    device_id,
                    start_time,
                    limit
                )
                return [SensorDataResponse(**dict(row)) for row in rows]
        except Exception as e:
            logger.error(f"Error fetching sensor history: {e}")
            raise
    
    async def get_device_stats(
        self,
        device_id: str,
        start_time: datetime
    ) -> Optional[PowerStatsResponse]:
        """Get statistical summary for a device"""
        try:
            async with self.pool.acquire() as conn:
                row = await conn.fetchrow(
                    """
                    SELECT 
                        device_id,
                        AVG(power_watts) as avg_power,
                        MAX(power_watts) as max_power,
                        MIN(power_watts) as min_power,
                        STDDEV(power_watts) as stddev_power,
                        COUNT(*) as sample_count
                    FROM sensor_data
                    WHERE device_id = $1 AND time >= $2
                    GROUP BY device_id
                    """,
                    device_id,
                    start_time
                )
                return PowerStatsResponse(**dict(row)) if row else None
        except Exception as e:
            logger.error(f"Error calculating device stats: {e}")
            raise
    
    async def get_alerts(
        self,
        resolved: Optional[bool] = None,
        limit: int = 100
    ) -> List[AlertResponse]:
        """Get system alerts"""
        try:
            async with self.pool.acquire() as conn:
                query = """
                    SELECT id, time, device_id, alert_type, severity,
                           message, threshold_value, actual_value,
                           resolved, resolved_at
                    FROM alerts
                """
                
                if resolved is not None:
                    query += " WHERE resolved = $1"
                    query += " ORDER BY time DESC LIMIT $2"
                    rows = await conn.fetch(query, resolved, limit)
                else:
                    query += " ORDER BY time DESC LIMIT $1"
                    rows = await conn.fetch(query, limit)
                
                return [AlertResponse(**dict(row)) for row in rows]
        except Exception as e:
            logger.error(f"Error fetching alerts: {e}")
            raise
    
    async def create_alert(self, alert_data: Dict[str, Any]) -> int:
        """Create a new alert"""
        try:
            async with self.pool.acquire() as conn:
                alert_id = await conn.fetchval(
                    """
                    INSERT INTO alerts 
                    (time, device_id, alert_type, severity, message,
                     threshold_value, actual_value)
                    VALUES ($1, $2, $3, $4, $5, $6, $7)
                    RETURNING id
                    """,
                    alert_data.get("time", datetime.utcnow()),
                    alert_data["device_id"],
                    alert_data["alert_type"],
                    alert_data.get("severity", "warning"),
                    alert_data["message"],
                    alert_data.get("threshold_value"),
                    alert_data.get("actual_value")
                )
                return alert_id
        except Exception as e:
            logger.error(f"Error creating alert: {e}")
            raise
    
    async def resolve_alert(self, alert_id: int) -> bool:
        """Mark an alert as resolved"""
        try:
            async with self.pool.acquire() as conn:
                result = await conn.execute(
                    """
                    UPDATE alerts
                    SET resolved = TRUE, resolved_at = $1
                    WHERE id = $2
                    """,
                    datetime.utcnow(),
                    alert_id
                )
                return result == "UPDATE 1"
        except Exception as e:
            logger.error(f"Error resolving alert: {e}")
            raise
    
    async def get_total_consumption(self, start_time: datetime) -> Dict[str, Any]:
        """Get total energy consumption across all devices using proper integration"""
        try:
            async with self.pool.acquire() as conn:
                # Use trapezoidal integration for accurate energy calculation
                row = await conn.fetchrow(
                    """
                    WITH time_series AS (
                        SELECT 
                            device_id,
                            time,
                            power_watts,
                            LEAD(time) OVER (PARTITION BY device_id ORDER BY time) as next_time,
                            LEAD(power_watts) OVER (PARTITION BY device_id ORDER BY time) as next_power
                        FROM sensor_data
                        WHERE time >= $1
                    ),
                    energy_per_device AS (
                        SELECT
                            device_id,
                            SUM(
                                (power_watts + COALESCE(next_power, power_watts)) / 2.0 / 1000.0 *
                                EXTRACT(EPOCH FROM (next_time - time)) / 3600.0
                            ) as kwh
                        FROM time_series
                        WHERE next_time IS NOT NULL
                        GROUP BY device_id
                    )
                    SELECT 
                        COALESCE(SUM(kwh), 0.0) as total_kwh,
                        COUNT(DISTINCT device_id) as device_count
                    FROM energy_per_device
                    """,
                    start_time
                )
                
                # Get current statistics
                stats = await conn.fetchrow(
                    """
                    SELECT 
                        AVG(power_watts) as avg_power_watts,
                        MAX(power_watts) as peak_power_watts
                    FROM sensor_data
                    WHERE time >= $1
                    """,
                    start_time
                )
                
                return {
                    "total_kwh": float(row["total_kwh"] or 0),
                    "avg_power_watts": float(stats["avg_power_watts"] or 0),
                    "peak_power_watts": float(stats["peak_power_watts"] or 0),
                    "device_count": int(row["device_count"] or 0),
                    "start_time": start_time,
                    "end_time": datetime.utcnow()
                }
        except Exception as e:
            logger.error(f"Error calculating total consumption: {e}")
            raise
    
    async def calculate_device_cost(
        self,
        device_id: str,
        start_time: datetime,
        end_time: datetime
    ) -> float:
        """Calculate cost for a specific device"""
        try:
            async with self.pool.acquire() as conn:
                cost = await conn.fetchval(
                    "SELECT calculate_cost($1, $2, $3)",
                    device_id,
                    start_time,
                    end_time
                )
                return float(cost or 0)
        except Exception as e:
            logger.error(f"Error calculating device cost: {e}")
            raise
    
    async def calculate_total_cost(
        self,
        start_time: datetime,
        end_time: datetime
    ) -> float:
        """Calculate total cost across all devices"""
        try:
            async with self.pool.acquire() as conn:
                # Get all devices and sum their costs
                devices = await conn.fetch("SELECT device_id FROM devices")
                total_cost = 0.0
                
                for device in devices:
                    cost = await conn.fetchval(
                        "SELECT calculate_cost($1, $2, $3)",
                        device["device_id"],
                        start_time,
                        end_time
                    )
                    total_cost += float(cost or 0)
                
                return total_cost
        except Exception as e:
            logger.error(f"Error calculating total cost: {e}")
            raise
